import datetime

class ProcurementLogisticsAgent:
    def verify_vendor_cyber_certification(self, vendor_metadata):
        """Operation: Verify_Vendor_Cyber_Certification - Compliance check."""
        # Checks for active SOC2 or ISO certifications
        return vendor_metadata.get("cert_status") == "Active"

    def detect_counterfeit_hardware_signatures(self, part_data):
        """Operation: Detect_Counterfeit_Hardware_Signatures - Profile check."""
        # Analyzes hardware signatures against known counterfeit patterns
        if "reconditioned" in part_data.get("source", "").lower():
            return "Risk: Potential Counterfeit Detected"
        return "Source Verified"

    def calculate_vendor_trust_decay(self, vendor_id, last_audit_date):
        """Operation: Calculate_Vendor_Trust_Decay - Temporal risk scoring."""
        # Enhancement: Automates trust score reduction for stale certifications
        days_since_audit = (datetime.date.today() - last_audit_date).days
        trust_score = max(0, 100 - (days_since_audit // 30) * 5)
        return trust_score
